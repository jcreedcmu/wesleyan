{\Gamma; \cdot \prov P \lol N}

{\Gamma_1,\Gamma_2 [P \lol N] \prov Q}

{\Gamma [N_1 \amp N_2] \prov Q}

\erule

{\Gamma  \prov [P_1 \oplus P_2]}
